SELECT nazwa FROM miejsca ORDER BY nazwa ASC

SELECT cena FROM miejsca WHERE nazwa = "Malbork"

SELECT nazwa, cena, link_obraz FROM miejsca WHERE link_obraz LIKE '0%'

SELECT nazwa, ROUND(AVG(liczba_doroslych)) FROM miejsca JOIN wycieczki ON wycieczki.id_miejsca = miejsca.id_miejsca GROUP BY miejsca.nazwa

    id_klienta INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
    imie VARCHAR(20),
    nazwisko VARCHAR(40),
    data_ur DATE);
