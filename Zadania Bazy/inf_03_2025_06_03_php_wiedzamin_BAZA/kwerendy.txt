SELECT `imie`,`nazwisko`,`klasa`,`data_wypozyczenia` FROM `uczniowie` WHERE `data_wypozyczenia` > "2020-03-02"

SELECT uczniowie.nazwisko, uczniowie.imie, uczniowie.data_wypozyczenia, ksiazki.kod, ksiazki.autor FROM uczniowie JOIN ksiazki ON ksiazki.id = uczniowie.id_ksiazki

DELETE FROM ksiazki WHERE autor LIKE '%0%';

SELECT autor, tytul, kod FROM ksiazki ORDER BY RAND() LIMIT 5;
