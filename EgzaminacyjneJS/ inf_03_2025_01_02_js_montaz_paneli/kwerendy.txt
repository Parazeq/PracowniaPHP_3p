SELECT imie, nazwisko FROM osoby WHERE imie LIKE 'A%';

SELECT zadanie, data FROM zadania WHERE zadanie LIKE '%mebli%' ORDER BY data ASC;

SELECT o.nazwisko, COUNT(z.id_zadania) AS "Liczba zadań" FROM osoby o JOIN zadania z ON o.id_osoba = z.osoba_id GROUP BY o.nazwisko;
